#include "../include/parser.h"

#include "../include/ui.h"

#define INITIAL_BUFFER_SIZE 4096
#define MAX_CPU_ENTRIES 33
#define MAX_PROCESSES 1024

// --- Struct definitions ---
typedef struct {
    pidStats* items;
    int count;
} ProcessList;

// FIX: A new struct to hold all data shared between threads
typedef struct {
    double cpu_usage[MAX_CPU_ENTRIES];
    memStats mem_info;
    ProcessInfo processed_list[MAX_PROCESSES];
    int num_total_cpu_entries;
    int num_processes;
} SharedData;

// --- Global variables for threading ---
static SharedData shared_data;
static pthread_mutex_t data_mutex;
static volatile int running = 1; // volatile tells the compiler this can change unexpectedly

// --- Function Prototypes ---
char *read_file_dynamically(const char *path);
ProcessList get_all_processes(void);
void* data_collector_thread(void* arg);


// ====================================================================
// The main function is now the UI THREAD
// ====================================================================
int main() {
    pthread_t data_thread_id;

    // Initialize mutex
    if (pthread_mutex_init(&data_mutex, NULL) != 0) {
        fprintf(stderr, "Mutex init failed\n");
        return 1;
    }

    // Create the data collector thread
    if (pthread_create(&data_thread_id, NULL, data_collector_thread, NULL) != 0) {
        fprintf(stderr, "Thread creation failed\n");
        return 1;
    }

    ui_init();

    // --- The UI Loop ---
    while (running) {
        // Handle input immediately
        int ch = getch();
        if (ch == 'q' || ch == 'Q') {
            running = 0; // Signal the other thread to stop
            continue;
        } else if (ch != ERR) {
            // No need to lock for this, as ui_handle_input only changes its own state
            ui_handle_input(ch, shared_data.num_processes);
        }

        // Create a local copy of data to draw, to minimize time holding the lock
        SharedData local_data_copy;
        pthread_mutex_lock(&data_mutex);
        local_data_copy = shared_data;
        pthread_mutex_unlock(&data_mutex);

        // Draw using the local copy
        ui_draw(local_data_copy.cpu_usage, &local_data_copy.mem_info, 
                local_data_copy.num_total_cpu_entries, 
                local_data_copy.processed_list, local_data_copy.num_processes);
        
        usleep(33000); // ~30 frames per second
    }

    // --- Cleanup ---
    pthread_join(data_thread_id, NULL); // Wait for the data thread to finish
    pthread_mutex_destroy(&data_mutex);
    ui_cleanup();
    return 0;
}

// ====================================================================
// This function is the DATA COLLECTOR THREAD
// ====================================================================
void* data_collector_thread(void* arg) {
    (void)arg; // Unused
    
    // Initialization for this thread
    cpuStat prevCpuStats[MAX_CPU_ENTRIES] = {0}, currCpuStats[MAX_CPU_ENTRIES] = {0};
    ProcessList prev_procs = { .items = NULL, .count = 0 };
    int num_cpu_entries = 0;

    char *initial_cpu_data = read_file_dynamically("/proc/stat");
    if (initial_cpu_data) {
        num_cpu_entries = cpuParser(initial_cpu_data, prevCpuStats, MAX_CPU_ENTRIES);
        free(initial_cpu_data);
    }
    prev_procs = get_all_processes();

    // The data collection loop
    while (running) {
        // Do all the heavy work...
        SharedData current_data;
        current_data.num_total_cpu_entries = num_cpu_entries;

        char* cpu_data = read_file_dynamically("/proc/stat");
        char* mem_data = read_file_dynamically("/proc/meminfo");
        ProcessList curr_procs = get_all_processes();

        if (cpu_data) {
            cpuParser(cpu_data, currCpuStats, MAX_CPU_ENTRIES);
            cpuUsage(prevCpuStats, currCpuStats, current_data.cpu_usage, num_cpu_entries);
        }
        if (mem_data) {
            // FIX 1: Replaced the stray character with a proper ampersand '&'
            // FIX 2: Corrected the typo from 't_data' to 'current_data'
            memParser(mem_data, &current_data.mem_info);
            free(mem_data);
        }

        long long total_cpu_time_delta = 0;
        if (cpu_data) {
            unsigned long long prev_total = prevCpuStats[0].user + prevCpuStats[0].nice + prevCpuStats[0].system + prevCpuStats[0].idle + prevCpuStats[0].iowait + prevCpuStats[0].irq + prevCpuStats[0].softirq;
            unsigned long long curr_total = currCpuStats[0].user + currCpuStats[0].nice + currCpuStats[0].system + currCpuStats[0].idle + currCpuStats[0].iowait + currCpuStats[0].irq + currCpuStats[0].softirq;
            total_cpu_time_delta = curr_total - prev_total;
        }

        for (int i = 0; i < curr_procs.count; ++i) {
            current_data.processed_list[i].stats = curr_procs.items[i];
            current_data.processed_list[i].cpu_percent = 0.0;
            for (int j = 0; j < prev_procs.count; ++j) {
                if (curr_procs.items[i].pid == prev_procs.items[j].pid) {
                    unsigned long long proc_time_delta = (curr_procs.items[i].utime + curr_procs.items[i].stime) - (prev_procs.items[j].utime + prev_procs.items[j].stime);
                    if (total_cpu_time_delta > 0) {
                        current_data.processed_list[i].cpu_percent = 100.0 * (double)proc_time_delta / (double)total_cpu_time_delta;
                    }
                    break;
                }
            }
            if (current_data.mem_info.memTotal > 0) {
                current_data.processed_list[i].mem_percent = 100.0 * (double)(current_data.processed_list[i].stats.rss * 4) / (double)current_data.mem_info.memTotal;
            }
        }
        current_data.num_processes = curr_procs.count;
        
        // Lock the mutex and copy the fresh data to the shared area
        pthread_mutex_lock(&data_mutex);
        shared_data = current_data;
        pthread_mutex_unlock(&data_mutex);

        // Update state for the next iteration
        if (cpu_data) {
            updateCpuState(prevCpuStats, currCpuStats, num_cpu_entries);
            free(cpu_data);
        }
        free(prev_procs.items);
        prev_procs = curr_procs;
        
        sleep(1); // Refresh data once per second
    }
    
    free(prev_procs.items); // Final cleanup for this thread
    return NULL;
}

// --- Helper Functions (unchanged) ---
// ... get_all_processes() and read_file_dynamically() are exactly the same as before ...

ProcessList get_all_processes() {
  DIR *proc_dir = opendir("/proc");
  if (!proc_dir)
    return (ProcessList){.items = NULL, .count = 0};

  ProcessList list = {.items = malloc(sizeof(pidStats) * MAX_PROCESSES),
                      .count = 0};
  if (!list.items) {
    closedir(proc_dir);
    return (ProcessList){.items = NULL, .count = 0};
  }

  struct dirent *entry;
  while ((entry = readdir(proc_dir)) != NULL && list.count < MAX_PROCESSES) {
    if (!isdigit(entry->d_name[0]))
      continue;
    char path[512];
    snprintf(path, sizeof(path), "/proc/%s/stat", entry->d_name);
    char *file_contents = read_file_dynamically(path);
    if (file_contents) {
      if (pidParser(file_contents, &list.items[list.count]))
        list.count++;
      free(file_contents);
    }
  }
  closedir(proc_dir);
  return list;
}

char *read_file_dynamically(const char *path) {
  int fd = open(path, O_RDONLY);
  if (fd < 0)
    return NULL;
  size_t buffer_size = INITIAL_BUFFER_SIZE;
  char *buffer = malloc(buffer_size);
  if (!buffer) {
    close(fd);
    return NULL;
  }
  size_t total_read = 0;
  while (1) {
    ssize_t bytes_read =
        read(fd, buffer + total_read, buffer_size - total_read - 1);
    if (bytes_read > 0) {
      total_read += bytes_read;
    } else if (bytes_read == 0) {
      break;
    } else {
      free(buffer);
      close(fd);
      return NULL;
    }
    if (total_read >= buffer_size - 1) {
      buffer_size *= 2;
      char *new_buffer = realloc(buffer, buffer_size);
      if (!new_buffer) {
        free(buffer);
        close(fd);
        return NULL;
      }
      buffer = new_buffer;
    }
  }
  buffer[total_read] = '\0';
  close(fd);
  return buffer;
}
